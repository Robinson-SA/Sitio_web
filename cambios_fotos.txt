+class FotoEmpleadoForm(forms.ModelForm):
+    class Meta:
+        model = Empleado
+        fields = ['foto']
+        widgets = {
+            'foto': forms.FileInput(attrs={'class': 'form-control', 'accept': 'image/png,image/jpeg'}),

+
+    def clean_foto(self):
+        foto = self.cleaned_data.get('foto')
+        if foto:
+            if foto.size > 5 * 1024 * 1024:
+                raise ValidationError('La imagen es demasiado grande. Máximo 5MB.')
+            allowed_mimes = ['image/jpeg', 'image/png', 'image/jpg']
+            if hasattr(foto, 'content_type') and foto.content_type not in allowed_mimes:
+                raise ValidationError('Tipo de archivo no permitido. Solo JPG o PNG.')
+        return foto

+@login_required(login_url='login')
+def agregar_foto_empleado(request, pk):
+    empleado = get_object_or_404(Empleado, pk=pk)
+    if empleado.foto:
+        messages.warning(request, f'"{empleado.nombre}" ya tiene una foto cargada.')
+        return redirect('lista_empleados')
+    if request.method == 'POST':
+        form = FotoEmpleadoForm(request.POST, request.FILES, instance=empleado)
+        if form.is_valid():
+            form.save()
+            messages.success(request, 'Foto agregada.')
+            return redirect('lista_empleados')
+    else:
+        form = FotoEmpleadoForm(instance=empleado)
+    return render(request, 'foto_empleado.html', {'form': form, 'empleado': empleado})
